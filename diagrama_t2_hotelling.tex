\node (start) [startstop] {Simulação da série $y_t$, $t=1\dots T$ de AR(1)};
\node (fit1) [process, below of=start] {Ajuste AR(1) em $y_t$ (obtendo $\hat{\boldsymbol\phi}$)};
\node (fit2) [process, below of=fit1] {Obtem $\hat {\phi}_{T+L} / [y_{L},\dots,y_{T+L-1}]$ };
\node (boot) [process, below of=fit2] {Boot $b=1\dots B$: Com $\hat{\boldsymbol\phi}$ gera-se nova série $y_t^*$ de tamanho $T$};
\node (calc) [process, below of=boot] {Obtém ${y_t}^{**} = y_t[L:T] \; y_t^*[1:L]$};
\node (fit3) [process, below of=calc] {Ajuste AR(1) em ${y_t}^{**}$ (obtém e guarda
    $\hat{\boldsymbol\phi^*}$)
};
\node (dist) [process, below of=fit3] {Fim do boot: intervalo de controle de $\hat{\boldsymbol\phi^*}$ L passos à frente};
\node (compare) [process, below of=dist] {Comparar a previsão $\hat {\phi}_{T+L}$ via AR(1) em $y_t$ aos quantis da distribuição};
\node (stop) [startstop, below of=compare] {Fim};
